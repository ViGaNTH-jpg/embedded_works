#define _XTAL_FREQ 4000000
#include <xc.h>
#include "picf877aconfig_1.h"

void main(void) 
{
 TRISBbits.TRISB1 = 1;   // RB1 como entrada
 TRISCbits.TRISC7 = 0;   // RC7 como salida

 PORTCbits.RC7 = 0;      // empieza el led apagado

 while (1)
 {
     if (PORTBbits.RB1 == 1)
     {
         PORTCbits.RC7 = 1;   // enciende led
     }

     if (PORTBbits.RB1 == 0)
     {
         PORTCbits.RC7 = 0;   // apaga el led
     }
 }
}
